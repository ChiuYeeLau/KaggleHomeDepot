plotDensity <- function(data.in, i) {
  data <- data.frame(x=data.in[,i], Target=data.in$target)
  p <- ggplot(data) +

doPlots <- function(data.in, fun, ii, ncol=3) {
  pp <- list()
  for (i in ii) {
    p <- fun(data.in=data.in, i=i)
    pp <- c(pp, list(p))
  }
  do.call("grid.arrange", c(pp, ncol=ncol))
}
